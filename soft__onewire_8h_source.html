115 /**
116  * @brief enable strong pullup
117  *
118  * This will drive the pin high. Useful for parasite powered devices, which temporary
119  * require high amount of current (which cannot be drawn though the bus pull-up resistor)
120  *
121  * @param a_bus bus descriptor
122  * @param a_enable 1 - enable pull-up, 0 - disable
123  */
124 #define sow_strong_pullup(__a_bus, __a_enable) \
125  if (__a_enable) { \
126  GPIO_CONFIGURE_AS_OUTPUT(&__a_bus->bus); \
127  GPIO_SET_HIGH(&__a_bus->bus); \
128  } \
129  else { \
130  GPIO_CONFIGURE_AS_INPUT(&__a_bus->bus); \
131  }
