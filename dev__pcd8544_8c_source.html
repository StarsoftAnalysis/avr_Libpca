208 void pcd8544_putc(struct dev_pcd8544_ctx *a_disp, char c) {
209  uint8_t font_length = sizeof(_font[0]);
210 
211  if (a_disp->y >= (PCD8544_H >> 3)) {
212  return;
213  }
214 
215  if ((a_disp->x + font_length + 1) > PCD8544_W || (c == '\n')) {
216  pcd8544_gotoxy(a_disp, 0, a_disp->y + 1);
217  }
218 
219  pcd8544_write(a_disp, PCD8544_DATA, 0x00);
220  for (uint8_t x = 0; x < font_length; x++ ) {
221  pcd8544_write(a_disp, PCD8544_DATA, pgm_read_byte(&_font[c - 0x20][x]));
222  }
223 
224  a_disp->x += (font_length + 1);
225 }
