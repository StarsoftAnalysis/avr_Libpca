72 /**
73  * @brief constructs the ENTRY_MODE SET Command byte for the display
74  *
75  * Keep in mind that all the those macros only assemble a valid command byte, THEY DO NOT SEND THE ACTUAL BYTE TO THE DISPLAY - you must do it yourself
76  * i.e. : hd44780_cmd(a_disp, HD44780_CMD_DISPLAY_CONTROL(1, 0, 0));
77  *
78  * @param __inc - cursor move direction 1 - increment, 0 - decrement
79  * @param __shift - 1 - shift display with every new character
80  */
81 #define HD44780_CMD_ENTRY_MODE(__inc, __shift) \
82  (0x04 | (__inc << 1) | __shift)
83 
84 /**
85  * @brief construct the DISPLAY Control command byte
86  *
87  * @param __onoff - turn the display on/off
88  * @param __cursos - turn the cursor on/off
89  * @param __blink - tunr on/off cursor blinking
90  */
91 #define HD44780_CMD_DISPLAY_CONTROL(__onoff, __cursor, __blink) \
92  (0x08 | (__onoff << 2 ) | (__cursor << 1) | __blink)
93 
94 /**
95  * @brief Constructs the CURSOR OR DISPLAY SHIFT command byte
96  *
97  * @param __moveshift - 1 - will shift the display by one in the specified direction, 0 - will move cursor instead
98  * @param __rightleft - 1 - move/shift right, 0 - move shift left
99  */
100 #define HD44780_CMD_CD_SHIFT(__moveshift, __rightleft) \
101  (0x10 | (__moveshift << 3) | (__rightleft << 2))
102 
103 /**
104  * @brief constructs the FUNCTION SET command byte
105  *
106  * @param __dl - sets interface data length 1 - 8 bit, 0 - 4 bit
107  * @param __lines - number of display lines 1 - 2 lines , 0 - 1 line
108  * @param __font - specifies display font 1 - 5x10, 0 - 5x8
109  */
110 #define HD44780_CMD_FUNCTION_SET(__dl, __lines, __font) \
111  (0x20 | (__dl << 4) | (__lines << 3) | (__font << 2))
112
