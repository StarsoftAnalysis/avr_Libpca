65 /**
66  * @brief Common interrupt service routine for all 3 timers
67  *
68  * @param sregs timer specific register & parameters structure
69  */
70 static void _isr_tdelay_handler(volatile struct regs *sregs) {
71 
72  if (!g_tc[sregs->timer].duration) {
73  // MASK the interrupt + disable the clock;
74  *(sregs->timsk) &= ~_BV(1); // OCIEXA
75  *(sregs->tccrb) &= 0xf8;
76 
77  // inverted logic if reset_cmpa_pin == 0, the pin will be zeroed
78  if (!g_tc[sregs->timer].reset_cmpa_pin)
79  *(sregs->port) &= ~_BV(sregs->pin);
80  }
81  else {
82  g_tc[sregs->timer].duration--;
83  }
84 }

86 /**
87  * @brief this function checks if the timer hasn't expired already
88  *
89  * The purpose is to protect the actual block function from blocking forever, once the
90  * timer already has expired and disabled it's interrupt
91  *
92  * @param a_timer timer
93  * @param timsk timer interrupt mask register
94  * @param map which bit to check
95  *
96  * @return 1 - if the timer hasn't expired and we should block, 0 otherwise
97  */
98 static uint8_t __tdc_block(e_timer a_timer, volatile uint8_t *timsk, uint8_t map) {
99  if (bit_is_clear((*timsk), map)) {
100  _tdc_set_duration(a_timer, 0x00);
101  return 0;
102  }
103  return 1;
104 }

243 void _tdc_block(e_tdelay_timer a_timer) {
244  unsigned char wait UNUSED = 0;
245 
246  // deadlock protection
247  switch (a_timer) {
248 #if TDELAY_IMPLEMENT_T0_INT == 1
249  case E_TDELAY_TIMER0:
250  wait = __tdc_block(a_timer, &TIMSK0, OCIE0A);
251  break;
252 #endif
253 
254 #if TDELAY_IMPLEMENT_T1_INT == 1
255  case E_TDELAY_TIMER1:
256  wait = __tdc_block(a_timer, &TIMSK1, OCIE1A);
257  break;
258 #endif
259 
260 #if TDELAY_IMPLEMENT_T2_INT == 1
261  case E_TDELAY_TIMER2:
262  wait = __tdc_block(a_timer, &TIMSK2, OCIE2A);
263  break;
264 #endif
265 
266  default:
267  break;
268  } // switch
269 
270 #if TDELAY_IMPLEMENT_T0_INT == 1 || TDELAY_IMPLEMENT_T1_INT == 1 || TDELAY_IMPLEMENT_T2_INT == 1
271  if (wait) while (g_tc[a_timer].duration);
272 #endif
273 }
