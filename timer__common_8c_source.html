73 uint32_t _timer_freq_prescale(e_timer a_timer, uint32_t a_freq, uint16_t a_criterion) {
74  uint8_t prescalers01[] = { 0x00, 0x03, 0x06, 0x08, 0x0a, 0x00 };
75  uint8_t prescalers2[] = { 0x00, 0x03, 0x05, 0x06, 0x07, 0x08, 0x0a, 0x00};
76 
77  /**
78  * combine both prescaler and ocr value in one 32bit value
79  */
80  uint32_t retval = 0x00;
81  uint16_t *ocr = (uint16_t *)&retval;
82  uint8_t *presc = ((uint8_t *)&retval) + 3;
83  uint8_t *prescalers = prescalers01;
84 
85  if (E_TIMER2 == a_timer) {
86  prescalers = prescalers2;
87  }
88 
89  do {
90  *ocr = (uint16_t) (F_CPU / ((a_freq << 1) * (0x01 << prescalers[*presc])));
91  ++*presc;
92  } while ((*ocr > a_criterion) && (prescalers[*presc]));
93 
94  --*ocr;
95  if (*ocr > a_criterion) *ocr = a_criterion;
96  return retval;
97 }
